JavaScript
//Canvas object
PolyGen.canvas = document.getElementById('canvas');
//Canvas context
PolyGen.context = PolyGen.canvas.getContext('2d');
//Thickness of side lines
PolyGen.thickness = 8;
//Color of lines. It can be in any CSS color format
PolyGen.color ='#000000';
//Size of shape
PolyGen.size = 409.6;
//Create by calling this function
PolyGen.create(5);
